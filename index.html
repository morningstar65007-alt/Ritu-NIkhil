⚡ JavaScript example

// creative function that greets by name + celebrates curiosity
function celebrateCuriosity(yourName) {
  const messages = [
    `✨ ${yourName}, your curiosity lights up the world! ✨`,
    `📚 Keep asking "why" — that's where genius begins.`,
    `🎨 Creativity + curiosity = magic formula! 🎨`
  ];
  const randomMsg = messages[Math.floor(Math.random() * messages.length)];
  console.log(randomMsg);
  return randomMsg;
}

// example usage: celebrateCuriosity("Jigyasa");
⚙️ JavaScript function that returns random encouragement — perfect for learning arrays, math, and functions!
